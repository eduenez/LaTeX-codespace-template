% monograph-template.tex — Main file for a book/monograph project (memoir class)
%
% Compile: latexmk (uses latexmkrc in this directory)
% Engine:  pdflatex
%
% To compile only specific chapters, uncomment and edit the \includeonly line.
% Do NOT remove or reorder the \include lines below.

\documentclass[letterpaper,12pt,oneside,openany]{memoir}

% ─── Shared preamble (submodule sty/, see sty/README.md) ────────────────────
\usepackage{di-base}
\usepackage{monograph}

% ─── Page Layout ─────────────────────────────────────────────────────────────
% memoir manages its own page geometry; avoid loading the geometry package
% alongside it, since the two layout systems can conflict.
\setlrmarginsandblock{3.5cm}{2.5cm}{*}
\setulmarginsandblock{3cm}{3cm}{*}
\checkandfixthelayout

% Chapter and heading style
\chapterstyle{madsen}
\nouppercaseheads
\pagestyle{ruled}

% ─── Bibliography ────────────────────────────────────────────────────────────
\usepackage{csquotes}
\usepackage[
  backend=biber,
  style=alphabetic,
  sorting=nyt,
  maxbibnames=10
]{biblatex}

% Local, monograph-specific bibliography (active by default).
\addbibresource{references.bib}

% Uncomment to also draw on the shared bibliography submodule (bib/, see
% bib/README.md for citation-key conventions and update instructions). Some
% coauthors may prefer to keep this monograph's references self-contained, or
% may be unfamiliar with git submodules — hence commented out by default.
% \addbibresource{bib/references.bib}

% ─── Compilation Control ────────────────────────────────────────────────────
% Uncomment the next line and list only the chapters you want to compile:
% \includeonly{ch-introduction,ch-background,ch-main-results,ch-conclusion}

% ─── Document Metadata ──────────────────────────────────────────────────────
\title{%
  \Huge\bfseries Title of the Monograph\\[0.5cm]
  \Large\itshape Subtitle or Series Information%
}
\author{Author One \and Author Two}
\date{\today}

% ═══════════════════════════════════════════════════════════════════════════
\begin{document}

\frontmatter

% Title page
\begin{titlingpage}
  \centering
  \vspace*{4cm}
  {\Huge\bfseries Title of the Monograph\par}
  \vspace{1cm}
  {\Large\itshape Subtitle or Series Information\par}
  \vspace{3cm}
  {\large Author One \quad Author Two\par}
  \vfill
  {\large \today\par}
\end{titlingpage}

\tableofcontents*
\clearpage

\mainmatter

% ─── Content Chapters ────────────────────────────────────────────────────────
% Each chapter lives in its own file in this directory.
% Use \include (not \input) so that \includeonly works.
\include{ch-introduction}
\include{ch-background}
\include{ch-main-results}
\include{ch-conclusion}

\backmatter

% ─── Bibliography ────────────────────────────────────────────────────────────
\printbibliography[heading=bibintoc]

\end{document}
