% monograph-template.tex — Main file for a book/monograph project (memoir class)
%
% Compile: latexmk (uses .latexmkrc at the project root)
% Engine:  pdfLaTeX
%
% To compile only specific chapters, uncomment and edit the \includeonly line.
% Do NOT remove or reorder the \include lines below.

\documentclass[letterpaper,12pt,oneside,openany]{memoir}

% ═════════════════════════════════════════════════════════════════════════════
% ┃ ALL \usepackage, macro, and environment declarations — including the memoir ┃
% ┃ page layout (\setlrmarginsandblock, \chapterstyle, …) — live in             ┃
% ┃ monograph-template.sty (loaded below), which loads the vendored shared      ┃
% ┃ packages from packages/ and then this project's additions. Do NOT add       ┃
% ┃ \usepackage / \newcommand / \newtheorem / layout commands here. This        ┃
% ┃ preamble carries ONLY the document class, that one style package, and       ┃
% ┃ semantic metadata (title / author / date). See STYLE.md.                    ┃
% ═════════════════════════════════════════════════════════════════════════════
\usepackage{monograph-template}

% ─── Compilation Control ────────────────────────────────────────────────────
% Uncomment the next line and list only the chapters you want to compile:
% \includeonly{monograph-template-1-introduction,monograph-template-2-background,monograph-template-3-main-results,monograph-template-4-conclusion}

% ─── Document Metadata ──────────────────────────────────────────────────────
\title{%
  \Huge\bfseries Title of the Monograph\\[0.5cm]
  \Large\itshape Subtitle or Series Information%
}
\author{Author One \and Author Two}
\date{\today}

% ═══════════════════════════════════════════════════════════════════════════════

%#
%# DOCUMENT: % \Huge\bfseries Title of the Monograph\\[0.5cm] \Large\itshape Subtitle or Series Information%
%#
\begin{document}

\frontmatter

% Title page
\begin{titlingpage}
  \centering
  \vspace*{4cm}
  {\Huge\bfseries Title of the Monograph\par}
  \vspace{1cm}
  {\Large\itshape Subtitle or Series Information\par}
  \vspace{3cm}
  {\large Author One \quad Author Two\par}
  \vfill
  {\large \today\par}
\end{titlingpage}

\tableofcontents*
\clearpage

\mainmatter

% ─── Content Chapters ────────────────────────────────────────────────────────
% Each chapter lives in its own file in this directory.
% Use \include (not \input) so that \includeonly works.
\include{monograph-template-1-introduction}
\include{monograph-template-2-background}
\include{monograph-template-3-main-results}
\include{monograph-template-4-conclusion}

\backmatter

% ─── Bibliography ────────────────────────────────────────────────────────────
% alpha style = [Doe25]-style labels, matching the group's other repositories.
\clearpage
\phantomsection
\addcontentsline{toc}{chapter}{\bibname}
\bibliographystyle{alpha}
\bibliography{references}

\end{document}
